SimpleRequestHandler examples

Post

var res = requestHandler.post(
  {id: $('#id').val(), name: $('#nameInput').val() }, // Parameters
	'/Controller/SetName'); // Action

Get

var res = requestHandler.get(
	{id: $('#id').val() }, // Parameters
	'/Controller/GetName'); // Action

Example get from jsontest.com

var res = requestHandler.get(
	undefined, // No parameters.
	'http://ip.jsontest.com/'); // Action

Example async get with success function callback

requestHandler.get(
						undefined, // No parameters.
						'http://ip.jsontest.com/', true, undefined, function(data) {
							$('#testSuccessFunctionCallback').toggle();
							$('#testSuccessFunctionRes').html('
Received IP:
' + data.ip + '
');
						}); // Action
